SDK system of options

Various parameters used in various steps from the picture acquisition to the marker reconstruction can be tuned. All the customisable settings are handled as options, which come in three differents types:

  1. integer options, which are 32 bits signed integer values;
  2. float options, which are single precision floating point values;
  3. data options, which are either strings or other binary information.

When using the C-style API (see C-style API), the options are accessed by the dedicated functions ftkGetInt32/ftkSetInt32, ftkGetFloat32/ftkSetFloat32, ftkGetData/ftkSetData, which requires the unique ID of the option. This unique ID is a uint32_t integer that is internally used to access the wanted option. There are no guarantee the ID of an option remains the same from one SDK version to the next. This is why the ftkEnumerateOptions function is provided: it allows to build a `mapping' between and option and its ID. For instance, getting the library version (a string option), is performed like this:

#include <ftkInterface.h>
#include <stdio.h>
#include <stdlib.h>
void deviceEnumCallback( uint64 sn, void* user, ftkDeviceType type )
{
uint64* lastDevice = (uint64*) user;
if ( lastDevice != 0uLL )
{
*lastDevice = sn;
}
}
void optionEnumCallback( uint64 sn, void* user, ftkOptionInfo* oi )
{
uint32* optionID = (uint32*) user;
if ( strcmp( oi->name, "Library Version" ) == 0 )
{
*optionID = oi->id;
}
}
main ()
{
ftkLibrary lib = ftkInit();
if ( lib == NULL )
{
printf( stderr, "Cannot initialise library\n" );
return EXIT_FAILURE;
}
uint64 sn = 0uLL;
if ( ftkEnumerateDevices( lib, deviceEnumCallback, &sn ) != FTK_OK )
{
printf( stderr, "Cannot enumerate devices\n" );
return EXIT_FAILURE;
}
else if ( sn == 0uLL )
{
printf( stderr, "No detected devices\n" );
return EXIT_FAILURE;
}
printf( stdout, "Detected device 0x" PRIx64 "\n", sn );
if ( ftkClose( &lib ) != FTK_OK )
{
printf( stderr, "Could not close the library\n" );
return EXIT_FAILURE;
}
uint32_t optionID = -1u;
if ( ftkEnumerateOptions( lib, sn, optionsEnumCallback, &optionID ) ==
FTK_OK && optionID != -1u )
{
ftkBuffer buffer;
if ( ftkGetInt32( id, sn, optionID, &buffer ) == FTK_OK )
{
printf( stdout, "Version is %s\n", buffer.data );
}
else
{
printf( stderr, "Could not read version\n" );
}
}
return EXIT_SUCCESS;
}

When using the C++ API (see C++ API), options are accessed by the dedicated methods atracsys::TrackingSystem::getIntOption / atracsys::TrackingSystem::setIntOption, atracsys::TrackingSystem::getFloatOption / atracsys::TrackingSystem::setFloatOption and atracsys::TrackingSystem::getDataOption / atracsys::TrackingSystem::setDataOption, which require either the option unique ID or the option name (i.e. a string). The same exercise as above reads:

#include "TrackingSystem.hpp"
#include <iostream>
int main()
{
atracsys::TrackingSystem tracker;
if ( tracker.initialise() != atracsys::Status::Ok )
{
tracker.streamLastError();
return 1;
}
if ( tracker.enumerateDevices() != atracsys::Status::Ok )
{
tracker.streamLastError();
return 1;
}
std::string versionStr;
if ( tracker.getDataOption( "Library Version", versionStr ) ==
atracsys::Status::Ok )
{
std::cout << "Version is " << versionStr << std::endl;
}
else
{
std::cerr << "Could not read version" << std::endl;
return 1;
}
return 0;
}
