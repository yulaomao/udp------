C-style API

The C-style API is the legacy interface, provided as a set of headers and a shared library. It is fully tested and validated. It can be used in a straightforward way with any compatible compiler.

Warning
The C API is not thread-safe. The user must ensure no concurrent calls to the SDK functions are performed.

The API consists of a set of functions, POD structures and callbacks. For instance, detecting the connected devices and get the serial number of the last detected one reads:

#include <ftkInterface.h>
#include <stdio.h>
#include <stdlib.h>
void deviceEnumCallback( uint64 sn, void* user, ftkDeviceType type )
{
uint64* lastDevice = (uint64*) user;
if ( lastDevice != 0uLL )
{
*lastDevice = sn;
}
}
main ()
{
ftkLibrary lib = ftkInit();
if ( lib == NULL )
{
printf( stderr, "Cannot initialise library\n" );
return EXIT_FAILURE;
}
uint64 sn = 0uLL;
if ( ftkEnumerateDevices( lib, deviceEnumCallback, &sn ) != FTK_OK )
{
printf( stderr, "Cannot enumerate devices\n" );
return EXIT_FAILURE;
}
else if ( sn == 0uLL )
{
printf( stderr, "No detected devices\n" );
return EXIT_FAILURE;
}
printf( stdout, "Detected device 0x" PRIx64 "\n", sn );
if ( ftkClose( &lib ) != FTK_OK )
{
printf( stderr, "Could not close the library\n" );
return EXIT_FAILURE;
}
return EXIT_SUCCESS;
}
